Groups Micro-Mvc handlers as controllers  

.. code-block:: php

    <?php

     $app = new \Phalcon\Mvc\Micro();
    
     $collection = new Collection();
    
     $collection->setHandler(new PostsController());
    
     $collection->get('/posts/edit/{id}', 'edit');
    
     $app->mount($collection);
