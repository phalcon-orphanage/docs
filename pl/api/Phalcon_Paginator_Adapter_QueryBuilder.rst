Pagination using a PHQL query builder as source of data  

.. code-block:: php

    <?php

     use Phalcon\Paginator\Adapter\QueryBuilder;
    
     $builder = $this->modelsManager->createBuilder()
                     ->columns('id, name')
                     ->from('Robots')
                     ->orderBy('name');
    
     $paginator = new QueryBuilder(
         [
             'builder' => $builder,
             'limit'   => 20,
             'page'    => 1,
         ]
     );
