This adapter allows to paginate data using a Phalcon\\Mvc\\Model resultset as a base.  

.. code-block:: php

    <?php

     use Phalcon\Paginator\Adapter\Model;
    
     $paginator = new Model(
         [
             'data'  => Robots::find(),
             'limit' => 25,
             'page'  => $currentPage
         ]
     );
    
     $paginate = $paginator->getPaginate();
